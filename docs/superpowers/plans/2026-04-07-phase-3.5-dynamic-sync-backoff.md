# Dynamic Sync Interval Backoff Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** NTPSyncEngine の sync interval を成功連続時に段階的に延長し、長尺録音時の SoC 熱負荷を低減する。

**Architecture:** `BackoffController` (純粋 struct) を NTPSyncEngine の private state として追加。runSyncLoop の sleep 時間を BackoffController 経由で決定。`resetBackoff()` public API を Facade の DriftMonitor jump 購読箇所から呼んで即時リセット。

**Tech Stack:** Swift 6.0 strict concurrency, Swift Testing, Swift Package Manager

**Spec:** `docs/superpowers/specs/2026-04-07-dynamic-sync-backoff-design.md`

```swift
import Foundation

/// 同期成功の連続回数に応じて sync interval を段階的に延長する純粋ロジック。
///
/// stages の先頭から順に進む。各段階で `promoteAfter` 回連続成功で次段へ昇格。
/// 失敗時は段階維持、successStreak のみリセット (頻繁な降格を避ける)。
/// jump 検出時など外部から `reset()` で初期段階に戻す。
struct BackoffController {
    let stages: [TimeInterval]
    let promoteAfter: Int

    private(set) var stageIndex: Int = 0
    private var successStreak: Int = 0

    init(stages: [TimeInterval], promoteAfter: Int) {
        precondition(!stages.isEmpty, "BackoffController.stages must not be empty")
        precondition(promoteAfter > 0, "BackoffController.promoteAfter must be > 0")
        self.stages = stages
        self.promoteAfter = promoteAfter
    }

    var currentInterval: TimeInterval { stages[stageIndex] }

    mutating func recordSuccess() {
        guard stageIndex < stages.count - 1 else {
            successStreak = 0
            return
        }
        successStreak += 1
        if successStreak >= promoteAfter {
            stageIndex += 1
            successStreak = 0
        }
    }

    mutating func recordFailure() {
        successStreak = 0
    }

    mutating func reset() {
        stageIndex = 0
        successStreak = 0
    }
}
```

```swift
import Testing
@testable import PeerClock

@Suite("BackoffController")
struct BackoffControllerTests {

    @Test("初期状態は stage 0")
    func initialStage() {
        let b = BackoffController(stages: [5, 10, 20, 30], promoteAfter: 3)
        #expect(b.stageIndex == 0)
        #expect(b.currentInterval == 5)
    }

    @Test("連続 promoteAfter 回成功で昇格")
    func promoteOnSuccess() {
        var b = BackoffController(stages: [5, 10, 20, 30], promoteAfter: 3)
        b.recordSuccess(); b.recordSuccess()
        #expect(b.stageIndex == 0)  // まだ 2 回
        b.recordSuccess()
        #expect(b.stageIndex == 1)
        #expect(b.currentInterval == 10)
        b.recordSuccess(); b.recordSuccess(); b.recordSuccess()
        #expect(b.stageIndex == 2)
        #expect(b.currentInterval == 20)
    }

    @Test("最終段で頭打ち")
    func saturateAtLastStage() {
        var b = BackoffController(stages: [5, 10], promoteAfter: 2)
        b.recordSuccess(); b.recordSuccess()
        #expect(b.stageIndex == 1)
        // さらに昇格しようとしても stay
        b.recordSuccess(); b.recordSuccess(); b.recordSuccess()
        #expect(b.stageIndex == 1)
        #expect(b.currentInterval == 10)
    }

    @Test("失敗は stage を維持し streak のみリセット")
    func failureKeepsStage() {
        var b = BackoffController(stages: [5, 10, 20], promoteAfter: 3)
        b.recordSuccess(); b.recordSuccess(); b.recordSuccess()  // → stage 1
        b.recordSuccess(); b.recordSuccess()                      // streak 2
        b.recordFailure()
        #expect(b.stageIndex == 1)  // 維持
        b.recordSuccess(); b.recordSuccess()
        #expect(b.stageIndex == 1)  // streak は 0 から再カウント、まだ 2
        b.recordSuccess()
        #expect(b.stageIndex == 2)
    }

    @Test("reset で初期段階へ")
    func resetReturnsToStageZero() {
        var b = BackoffController(stages: [5, 10, 20, 30], promoteAfter: 2)
        b.recordSuccess(); b.recordSuccess()  // → 1
        b.recordSuccess(); b.recordSuccess()  // → 2
        #expect(b.stageIndex == 2)
        b.reset()
        #expect(b.stageIndex == 0)
        #expect(b.currentInterval == 5)
    }

    @Test("単一段では昇格しない")
    func singleStage() {
        var b = BackoffController(stages: [5], promoteAfter: 3)
        b.recordSuccess(); b.recordSuccess(); b.recordSuccess(); b.recordSuccess()
        #expect(b.stageIndex == 0)
        #expect(b.currentInterval == 5)
    }
}
```

```swift
    private func runSyncLoop() async {
        while !Task.isCancelled {
            guard let coordinatorID = lock.withLock({ self.coordinatorID }) else { break }

            let measurements = await collectMeasurements(coordinator: coordinatorID)

            let interval: TimeInterval
            if measurements.isEmpty {
                // 測定失敗 — 段階維持、streak リセット
                interval = lock.withLock {
                    self.backoff.recordFailure()
                    return self.backoff.currentInterval
                }
            } else {
                let filtered = NTPSyncEngine.bestHalfFilter(measurements)
                let offsetNs = NTPSyncEngine.meanOffset(filtered)
                let offsetSeconds = offsetNs / 1_000_000_000.0

                lock.withLock { self._currentOffset = offsetSeconds }

                let bestDelay = filtered.min(by: { $0.delay < $1.delay })?.delay ?? 0
                let quality = SyncQuality(
                    offsetNs: Int64(offsetNs),
                    roundTripDelayNs: bestDelay,
                    confidence: min(1.0, Double(filtered.count) / Double(configuration.syncMeasurements))
                )
                syncStateContinuation.yield(.synced(offset: offsetSeconds, quality: quality))

                interval = lock.withLock {
                    self.backoff.recordSuccess()
                    return self.backoff.currentInterval
                }
            }

            do {
                try await Task.sleep(for: .seconds(interval))
            } catch {
                break
            }
        }
    }
```

```swift
    @Test("resetBackoff() returns engine to first stage")
    func resetBackoffPublicAPI() async throws {
        let network = MockNetwork()
        let coordinator: PeerID = "A"
        let follower: PeerID = "B"

        let config = Configuration(
            syncBackoffStages: [0.05, 0.5, 1.0],
            syncBackoffPromoteAfter: 2,
            syncMeasurements: 2,
            syncMeasurementInterval: 0.005
        )

        let coordTransport = network.createTransport(for: coordinator)
        let followTransport = network.createTransport(for: follower)
        // coordinator side responder
        let coordRouter = CommandRouter(transport: coordTransport, localPeerID: coordinator)
        let _ = await coordRouter.start()
        let respond = Task {
            for await (peer, msg) in await coordRouter.syncMessages {
                if case .syncRequest(let t0) = msg.payload {
                    let t1 = currentMonotonicNs()
                    let t2 = currentMonotonicNs()
                    let resp = Message(
                        category: .syncResponse,
                        flags: 0,
                        payload: .syncResponse(t0: t0, t1: t1, t2: t2)
                    )
                    try? await coordTransport.sendUnreliable(MessageCodec.encode(resp), to: peer)
                }
            }
        }

        let followRouter = CommandRouter(transport: followTransport, localPeerID: follower)
        let _ = await followRouter.start()
        let engine = NTPSyncEngine(
            transport: followTransport,
            localPeerID: follower,
            configuration: config,
            syncMessageStream: await followRouter.syncMessages
        )
        await engine.start(coordinator: coordinator)

        // 数サイクル待って昇格させる
        try await Task.sleep(for: .milliseconds(400))
        engine.resetBackoff()
        // resetBackoff は struct 内部状態のリセットのみ。
        // クラッシュせず後続ループが続くことを確認する程度で十分。
        try await Task.sleep(for: .milliseconds(100))

        await engine.stop()
        respond.cancel()
    }
```

注: 上記は `currentMonotonicNs()` など既存ヘルパに依存。NTPSyncEngineTests に類似コードがあれば真似る (Step 1 で確認)。もし API シグネチャが合わなければ、テストは省略して BackoffController 単体テスト (Task 1) のカバレッジで済ませる判断も可。
