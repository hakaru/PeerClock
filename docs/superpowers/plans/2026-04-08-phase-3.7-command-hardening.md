> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** `CommandRouter` の `syncMessages` を `syncRequests`/`syncResponses` に分割して multiplexing race を解消し、`commandBroadcast`/`commandUnicast` に `commandID + logicalVersion + senderID` を追加して重複排除と前進保証を実装する。

**Tech Stack:** Swift 6.0 strict concurrency, Swift Testing, Swift Package Manager

```swift
    // MARK: - v2 Helpers

    private static func encodeUUID(_ uuid: UUID) -> [UInt8] {
        let b = uuid.uuid
        return [b.0, b.1, b.2, b.3, b.4, b.5, b.6, b.7,
                b.8, b.9, b.10, b.11, b.12, b.13, b.14, b.15]
    }

    private static func decodeUUID(_ data: Data, offset: Int) throws -> UUID {
        guard data.count >= offset + 16 else {
            throw MessageCodecError.truncatedData
        }
        let b = data.subdata(in: offset..<(offset + 16))
        let bytes = [UInt8](b)
        return UUID(uuid: (
            bytes[0], bytes[1], bytes[2], bytes[3],
            bytes[4], bytes[5], bytes[6], bytes[7],
            bytes[8], bytes[9], bytes[10], bytes[11],
            bytes[12], bytes[13], bytes[14], bytes[15]
        ))
    }

    /// Decode [16B UUID][8B ver][PeerID][Command] payload.
    private static func decodeIdentifiedCommand(
        _ payload: Data
    ) throws -> (UUID, UInt64, PeerID, Command) {
        let cid = try decodeUUID(payload, offset: 0)
        let ver = decodeUInt64(payload, offset: 16)
        // PeerID is 16 bytes (UUID-based), see PeerID.data
        guard payload.count >= 16 + 8 + 16 else {
            throw MessageCodecError.truncatedData
        }
        let senderBytes = payload.subdata(in: 24..<40)
        let sender = try PeerID(data: senderBytes)
        let commandData = payload.subdata(in: 40..<payload.count)
        let command = try decodeCommand(commandData)
        return (cid, ver, sender, command)
    }
```

**注意**: 上記 `PeerID.data` サイズと `PeerID(data:)` の signature は実装時に `Sources/PeerClock/Types.swift` の PeerID 定義で要確認。UUID ベースなら 16 bytes のはず。decodeUInt64 の signature も同様。もし既存 helper `decodeUInt64(_ data: Data, offset: Int) -> UInt64` でなければ既存スタイルに合わせる。

```swift
import Foundation

/// Transport 経由でピア間のアプリケーションコマンドを送受信するルーター。
///
/// identity: 送信時に commandID (UUID) と logicalVersion (単調増加) を自動付与。
/// 受信時は (sender, commandID) の重複を排除し、同 sender の version 前進を保証する。
public final class CommandRouter: CommandHandler, @unchecked Sendable {

    private let transport: any Transport
    private let localPeerID: PeerID
    private let lock = NSLock()

    private var listenTask: Task<Void, Never>?
    private var commandContinuation: AsyncStream<(PeerID, Command)>.Continuation?
    private var syncRequestsContinuation: AsyncStream<(PeerID, Message)>.Continuation?
    private var syncResponsesContinuation: AsyncStream<(PeerID, Message)>.Continuation?
    private var heartbeatContinuation: AsyncStream<PeerID>.Continuation?
    private var statusPushContinuation: AsyncStream<(PeerID, UInt64, [StatusEntry])>.Continuation?

    // Identity state (lock 配下)
    private var nextLogicalVersion: UInt64 = 1
    private var seenCommands: [PeerID: Set<UUID>] = [:]
    private var seenOrder: [PeerID: [UUID]] = [:]
    private var highestVersionPerSender: [PeerID: UInt64] = [:]
    private let maxSeenPerSender: Int

    public let incomingCommands: AsyncStream<(PeerID, Command)>
    /// PING メッセージ (SYNC_REQUEST) のみを流すストリーム。
    let syncRequests: AsyncStream<(PeerID, Message)>
    /// PONG メッセージ (SYNC_RESPONSE) のみを流すストリーム。
    let syncResponses: AsyncStream<(PeerID, Message)>
    let heartbeatSenders: AsyncStream<PeerID>
    let statusPushes: AsyncStream<(PeerID, UInt64, [StatusEntry])>

    public init(transport: any Transport, localPeerID: PeerID, maxSeenPerSender: Int = 1000) {
        self.transport = transport
        self.localPeerID = localPeerID
        self.maxSeenPerSender = maxSeenPerSender

        var cmdCont: AsyncStream<(PeerID, Command)>.Continuation!
        incomingCommands = AsyncStream { cmdCont = $0 }
        commandContinuation = cmdCont

        var reqCont: AsyncStream<(PeerID, Message)>.Continuation!
        syncRequests = AsyncStream { reqCont = $0 }
        syncRequestsContinuation = reqCont

        var respCont: AsyncStream<(PeerID, Message)>.Continuation!
        syncResponses = AsyncStream { respCont = $0 }
        syncResponsesContinuation = respCont

        var hbCont: AsyncStream<PeerID>.Continuation!
        heartbeatSenders = AsyncStream { hbCont = $0 }
        heartbeatContinuation = hbCont

        var spCont: AsyncStream<(PeerID, UInt64, [StatusEntry])>.Continuation!
        statusPushes = AsyncStream { spCont = $0 }
        statusPushContinuation = spCont

        startListening()
    }

    public func send(_ command: Command, to peer: PeerID) async throws {
        let (id, ver) = nextIdentity()
        let msg = Message.commandUnicast(
            commandID: id, logicalVersion: ver,
            senderID: localPeerID, command: command
        )
        try await transport.send(MessageCodec.encode(msg), to: peer)
    }

    public func broadcast(_ command: Command) async throws {
        let (id, ver) = nextIdentity()
        let msg = Message.commandBroadcast(
            commandID: id, logicalVersion: ver,
            senderID: localPeerID, command: command
        )
        try await transport.broadcast(MessageCodec.encode(msg))
    }

    /// Clear all state for a disconnected peer. Called from PeerClock heartbeat handler.
    public func forgetPeer(_ peer: PeerID) {
        lock.withLock {
            seenCommands[peer] = nil
            seenOrder[peer] = nil
            highestVersionPerSender[peer] = nil
        }
    }

    private func nextIdentity() -> (UUID, UInt64) {
        lock.withLock {
            let v = nextLogicalVersion
            nextLogicalVersion += 1
            return (UUID(), v)
        }
    }

    /// Returns true if the command should be delivered to incomingCommands.
    /// Updates internal state. Caller must have decoded the command identity.
    private func shouldDeliver(sender: PeerID, commandID: UUID, version: UInt64) -> Bool {
        lock.withLock {
            // 1. Duplicate check
            if seenCommands[sender, default: []].contains(commandID) {
                return false
            }
            // 2. Version forward check
            if let highest = highestVersionPerSender[sender], version < highest {
                return false
            }
            // 3. Accept: update state
            var seen = seenCommands[sender, default: []]
            var order = seenOrder[sender, default: []]
            seen.insert(commandID)
            order.append(commandID)
            if order.count > maxSeenPerSender {
                let evict = order.removeFirst()
                seen.remove(evict)
            }
            seenCommands[sender] = seen
            seenOrder[sender] = order
            highestVersionPerSender[sender] = max(
                highestVersionPerSender[sender] ?? 0, version
            )
            return true
        }
    }

    private func startListening() {
        listenTask = Task { [weak self] in
            guard let self else { return }
            for await (peer, data) in self.transport.incomingMessages {
                guard let message = try? MessageCodec.decode(data) else { continue }
                switch message {
                case .commandBroadcast(let cid, let ver, let sender, let command),
                     .commandUnicast(let cid, let ver, let sender, let command):
                    if self.shouldDeliver(sender: sender, commandID: cid, version: ver) {
                        self.commandContinuation?.yield((sender, command))
                    }
                case .ping:
                    self.syncRequestsContinuation?.yield((peer, message))
                case .pong:
                    self.syncResponsesContinuation?.yield((peer, message))
                case .heartbeat:
                    self.heartbeatContinuation?.yield(peer)
                case .statusPush(_, let generation, let entries):
                    self.statusPushContinuation?.yield((peer, generation, entries))
                default:
                    break
                }
            }
        }
    }

    deinit {
        listenTask?.cancel()
        commandContinuation?.finish()
        syncRequestsContinuation?.finish()
        syncResponsesContinuation?.finish()
        heartbeatContinuation?.finish()
        statusPushContinuation?.finish()
    }
}
```

import Foundation
@testable import PeerClock

@Suite("CommandRouter — Identity & Dedup")
struct CommandRouterIdentityTests {

    @Test("send() attaches commandID and monotonic logicalVersion")
    func sendAttachesIdentity() async throws {
        let network = MockNetwork()
        let senderID = PeerID(UUID())
        let peerID = PeerID(UUID())
        let senderTransport = network.createTransport(for: senderID)
        let peerTransport = network.createTransport(for: peerID)
        let senderRouter = CommandRouter(transport: senderTransport, localPeerID: senderID)

        // Collect 2 commands

        let recvTask = Task {
            var count = 0
            for await item in peerRouter.incomingCommands {
                received.append(item)
                count += 1
                if count >= 2 { break }
            }
        }

        try await senderRouter.send(Command(type: "a", payload: Data()), to: peerID)
        try await senderRouter.send(Command(type: "b", payload: Data()), to: peerID)
        try await Task.sleep(for: .milliseconds(200))

        #expect(items.count == 2)
        #expect(items[0].1.type == "a")

        // but then commandIDs differ. To test dedup we must control the commandID:
        // Alternative: call shouldDeliver directly via @testable on a same peerRouter
